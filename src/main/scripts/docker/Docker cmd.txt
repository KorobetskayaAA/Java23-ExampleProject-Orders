# Запускать из директории, где лежит compose.yaml!

# Создать и запустить контейнеры в командной строке
docker compose up --build

# Создать и запустить контейнеры в фоновом режиме
docker compose up --build -d

# Остановить и удалить контейнеры и их тома (сохранить данные)
docker compose down

# Остановить и удалить контейнеры и их тома (полностью стереть все данные)
docker compose down -v
